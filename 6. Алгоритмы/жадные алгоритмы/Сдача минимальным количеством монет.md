function minCoins(coins, amount) {
    coins.sort((a, b) => b - a); // Жадно берем самые крупные монеты
    let count = 0;
    for (let coin of coins) {
        while (amount >= coin) {
            amount -= coin;
            count++;
        }
    }
    return amount === 0 ? count : -1; // Если не удалось собрать сумму
}

console.log(minCoins([1, 5, 10, 25], 63)); // 6 (25+25+10+1+1+1)
