import { User } from './entities/user.entity';
import { CreateUserInput } from './dto/create-user.dto';
import { UserService } from './user.service';

@Resolver(() => User)
export class UserResolver {
  constructor(private readonly userService: UserService) {}

  @Query(() => [User], { name: 'users' })
  findAll(): User[] {
    return this.userService.findAll();

  @Mutation(() => User)

### **Резюме**

* **GraphQLModule** позволяет легко настроить GraphQL в NestJS.
* Резолверы (`Resolvers`) обрабатывают запросы и связаны с сервисами.
* Генерация схемы (`schema.gql`) автоматизирует процесс, упрощая документацию.
* Благодаря мощной поддержке типов TypeScript, разработка GraphQL API в NestJS становится предсказуемой, гибкой и безопасной.

Теперь у вас есть базовый GraphQL API для управления сущностью **User**! 🎉
